92 const unsigned int n_pad_x, const unsigned int n_pad_y) {
93 canvas_name = std::string("c_") + canvas_name;
94 TCanvas* c = rh::CreateCanvas(canvas_name.c_str(), canvas_name.c_str(), n_pad_x, n_pad_y);
95
96 TF1* f_wave = new TF1(
97 "f_wave",
98 [](const double* x, const double* p) {
99 return p[0] * TMath::Sin(TMath::TwoPi() * p[1] * x[0]) + p[2];
