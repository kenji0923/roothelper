95template <class ObjectType>
96int ObjectList::LoadData(TDirectory* directory, const std::vector<std::string>& path_list,
97 const std::vector<std::string>& title_list) {
98 const int n_obj = path_list.size();
99
100 for (int i_obj = 0; i_obj < n_obj; ++i_obj) {
101 ObjectType* obj_buffer = nullptr;
102 directory->GetObject(path_list[i_obj].c_str(), obj_buffer);
103
104 if (obj_buffer == nullptr) {
105 throw std::runtime_error(path_list[i_obj] + " was not found");
106 }
107
108 object_list_.emplace_back(obj_buffer);
109 directory_list_.emplace_back(directory);
110 path_list_.emplace_back(path_list[i_obj]);
111
112 if (i_obj < title_list.size()) {
113 static_cast<TNamed*>(obj_buffer)->SetTitle(title_list[i_obj].c_str());
114 title_list_.emplace_back(title_list[i_obj]);
115 } else {
116 title_list_.emplace_back(object_list_.back()->GetTitle());
117 }
118 }
119
120 return n_obj;
121}
